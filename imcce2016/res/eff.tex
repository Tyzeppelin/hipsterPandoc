\begin{algorithm}
  \label{alg:splitting}
  \caption{Algorithme de séparation des facteurs irr\'eductibles de multiplicité > 1}
  \begin{algorithmic}[1]
    \Require{f un polynôme f(x) unitaire et sans racines multiples dans $\mathbb{F}_p[x]$, avec p un nombre premier, 
            produit de d facteurs de degré n $f(x) = \displaystyle\prod_{i=1}^{d}f_i(x)$ }
    \Ensure{f de la forme}
    \Statex
    \Function{equal\_degree\_splitting}{$f, n$}
      \If{$deg(f(x)) \leq n$}
        \State \Return{$\{f(x)\}$}
      \EndIf
      \State $m \gets deg(f(x))/n $
      \State $S \gets \{f(x)\}$
      \While{$SizeOf(S) < m$}
        \State $v(x) \gets PolyAleatoire(degre \beq 2n-1)$
        \State $v(x) \gets v(x)^{(q^n-1)/2}-1$
        \State $g(x) \gets gcd(f(x), v(x))$
        \If{$g(x) \neq 1$ and $g(x) \neq f(x)$}
          \State $S \gets equal\_degree\_splitting(g(x), n) \cup equal\_degree\_splitting(f(x)/g(x), n)$
        \EndIf
      \EndWhile
      \State \Return{$S$}
    \EndFunction
  \end{algorithmic}
\end{algorithm}
