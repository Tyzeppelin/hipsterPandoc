\begin{algorithm}
  \caption{Algorithme de Yun - Factorisation des facteurs de multiplicit\'e > 1}
  \begin{algorithmic}[1]
    \Require{f un polynôme unitaire dans $\mathbb{F}_p[x]$ et p un nombre premier}
    \Ensure{L'ensemble $S(f^{\star}, i)$ des polynômes $f^{\star}(x)$ sans racines multiples et leur multiplicité $i$}
    \Statex
    \Function{squarefree\_factorization}{$f$}
      \State $i \gets 1$
      \State $S \gets \{\}$
      \State $h(x) \gets f'(x)$
      \If{$h == 0$}
        \State $f(x) \gets f(x)^{1/p}$
        \State $S \gets \cup \{squarefree\_factorization(f(x)), p\}$
      \Else
        \State $c(x) \gets gcd(f(x), h(x))$
        \State $w(x) \gets f(x)/c(x)$
        \While{$w(x) \neq 1$}
          \State $y(x) \gets gcd(w(x), c(x))$
          \State $f^{\star}(x) \gets w(x)/y(x)$
          \State $S \gets S \cup \{f^{\star} (x), i\}$
          \State $i \gets i+1$
          \State $w(x) \gets y(x)$
          \State $c(x) \gets c(x)/y(x)$
        \EndWhile
        \If{$c(x) \neq 1$}
          \State $c(x) \gets c(x)^{1/p}$
          \State $S \gets S \cup \{(squarefree\_factorization(c(x)),p)\} $
        \EndIf
      \EndIf
      \State \Return{$S$}
    \EndFunction
  \end{algorithmic}
\end{algorithm}
